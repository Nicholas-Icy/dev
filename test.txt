package com.example.loginvalidator.controller;

import com.example.loginvalidator.model.User;
import com.example.loginvalidator.repository.UserRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
public class LoginController {

    @Autowired
    private UserRepository repo;

    @GetMapping("/")
    public String showLogin(Model model) {
        model.addAttribute("user", new User());
        return "login";
    }

    @PostMapping("/validate")
    public String validate(@ModelAttribute User user, Model model) {
        User dbUser = repo.findById(user.getUsername()).orElse(null);

        if (dbUser != null && dbUser.getPassword().equals(user.getPassword())) {
            model.addAttribute("message", "Login Successful");
        } else {
            model.addAttribute("message", "Invalid Credentials");
        }

        return "result";
    }
}

5. Controller Class (MobileController.java)
package com.example.mobilevalidator.controller;

import com.example.mobilevalidator.model.Mobile;
import com.example.mobilevalidator.repository.MobileRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
public class MobileController {

    @Autowired
    private MobileRepository repo;

    @GetMapping("/")
    public String showForm(Model model) {
        model.addAttribute("mobile", new Mobile());
        return "form";
    }

    @PostMapping("/validate")
    public String validate(@ModelAttribute Mobile mobile, Model model) {

        String num = mobile.getNumber();

        // Validation: must be exactly 10 digits
        if (num.matches("\\d{10}")) {
            repo.save(mobile);
            model.addAttribute("message", "Valid Mobile Number (Saved)");
        } else {
            model.addAttribute("message", "Invalid Mobile Number");
        }

        return "result";
    }
}

5. Controller Class (StudentController.java)
package com.example.studentgreeting.controller;

import com.example.studentgreeting.model.Student;
import com.example.studentgreeting.repository.StudentRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
public class StudentController {

    @Autowired
    private StudentRepository repo;

    @GetMapping("/")
    public String showForm(Model model) {
        model.addAttribute("student", new Student());
        return "form";
    }

    @PostMapping("/greet")
    public String greet(@ModelAttribute Student student, Model model) {

        repo.save(student);

        model.addAttribute("message", "Hello, " + student.getName() + "!");
        return "result";
    }

    @GetMapping("/view")
    public String viewAll(Model model) {
        model.addAttribute("students", repo.findAll());
        return "view";
    }
}

package com.example.studentgreeting.controller;

import com.example.studentgreeting.model.Student;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

import java.util.ArrayList;
import java.util.List;

@Controller
public class StudentController {

    // In-memory storage
    private List<Student> studentList = new ArrayList<>();

    @GetMapping("/")
    public String showForm(Model model) {
        model.addAttribute("student", new Student());
        return "form";
    }

    @PostMapping("/greet")
    public String greet(@ModelAttribute Student student, Model model) {

        studentList.add(student);

        model.addAttribute("message", "Hello, " + student.getName() + "!");
        return "result";
    }

    @GetMapping("/view")
    public String viewAll(Model model) {
        model.addAttribute("students", studentList);
        return "view";
    }
}
